### for

for((i=0;i<10;i++))
do
    echo $i
done

for i in {1..10}
do
    echo $i
done

for i in $(seq 1 10)
do
    echo $i
done

此处无差别，均为独立个体
for i in $*
for i in $@

此处有差别，
"$*" = "$1$2$3$4...$n"
"$@" = "$1" "$2" "$3" "$4" ... "$n"

for i in "$*"
for i in "$@"



数组循环


除了用${array[@]}直接遍历数组的元素值

还可以用${!array[@]}、${!array[*]} 获取下标数组进行循环


```
[root@master 20260611]# array_1=(a bb ccc ddd)
[root@master 20260611]# echo ${!array_1[*]}
0 1 2 3
[root@master 20260611]# echo ${!array_1[@]}
0 1 2 3

```
